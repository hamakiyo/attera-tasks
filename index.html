/** * @fileoverview タスク管理アプリ用 バックエンドAPI (v2: メンバー管理＆Slack連携追加) */ const CONFIG = { SHEET_NAME_DB: 'TaskMaster', SHEET_NAME_MEMBERS: 'Members', TIMEZONE: 'Asia/Tokyo', DATE_FORMAT: 'yyyy/MM/dd HH:mm:ss', // ★ここにSlackのWebhook URLを貼り付けてください★ SLACK_WEBHOOK_URL: "" }; // ========================================== // API エンドポイント (GET: データの取得) // ========================================== function doGet(e) { try { const { tasks } = getDatabase_(CONFIG.SHEET_NAME_DB); const { tasks: members } = getDatabase_(CONFIG.SHEET_NAME_MEMBERS); return ContentService.createTextOutput(JSON.stringify({ status: 'success', data: { tasks: tasks, members: members } })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ========================================== // API エンドポイント (POST: データの更新・追加) // ========================================== function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const action = payload.action; let result = {}; switch (action) { case 'addTask': result = addTask_(payload.data); break; case 'updateStatus': result = updateTaskStatus_(payload.data.taskId, payload.data.newStatus); break; case 'updateTask': result = updateTaskDetails_(payload.data); break; case 'addMember': result = addMember_(payload.data); break; case 'sendSlack': result = sendSlackNotification_(payload.data.userId); break; default: throw new Error('不明なアクションです: ' + action); } return ContentService.createTextOutput(JSON.stringify({ status: 'success', result: result })).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON); } } // ========================================== // コア・ロジック (内部処理) // ========================================== // 汎用データ取得関数 function getDatabase_(sheetName) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) throw new Error(`${sheetName} シートが存在しません。`); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { tasks: [], sheet, headers: data[0] || [] }; const headers = data[0]; const body = data.slice(1); const tasks = body.map(row => { let obj = {}; headers.forEach((h, i) => { if (row[i] instanceof Date) { const d = row[i]; if(d.getHours() === 0 && d.getMinutes() === 0 && d.getSeconds() === 0) { obj[h] = Utilities.formatDate(d, CONFIG.TIMEZONE, 'yyyy-MM-dd'); } else { obj[h] = Utilities.formatDate(d, CONFIG.TIMEZONE, CONFIG.DATE_FORMAT); } } else { obj[h] = row[i]; } }); return obj; }); return { tasks, sheet, headers }; } // 新規メンバー追加 function addMember_(memberData) { const { sheet, headers } = getDatabase_(CONFIG.SHEET_NAME_MEMBERS); const now = new Date(); const newRow = headers.map(header => { switch(header) { case 'User_ID': return memberData.userId; case 'User_Name': return memberData.userName; case 'Color': return memberData.color || 'blue'; case 'Created_At':return now; default: return ''; } }); sheet.appendRow(newRow); return { message: 'メンバーを追加しました' }; } // Slack手動通知ロジック (役割: 抽出係 兼 POST係) function sendSlackNotification_(userId) { if(!CONFIG.SLACK_WEBHOOK_URL) throw new Error("Slack Webhook URLが設定されていません。"); const { tasks } = getDatabase_(CONFIG.SHEET_NAME_DB); const todayStr = Utilities.formatDate(new Date(), CONFIG.TIMEZONE, 'yyyy-MM-dd'); // 指定ユーザー(allの場合は全員)のタスクをフィルタリング let targetTasks = tasks.filter(t => t.Status !== 'done'); if(userId !== 'all') { targetTasks = targetTasks.filter(t => t.User_ID === userId); } const pastDue = targetTasks.filter(t => t.Deadline < todayStr); const todayTask = targetTasks.filter(t => t.Deadline === todayStr); let msg = `*[手動通知] 現在のタスク状況 (${userId === 'all' ? '全体' : userId})*\n\n`; if(pastDue.length > 0) { msg += `*🚨 期限超過*\n`; pastDue.forEach(t => msg += `• <@${t.User_ID}>: ${t.Task_Name} (${t.Deadline})\n`); msg += `\n`; } if(todayTask.length > 0) { msg += `*☀️ 本日やるべきタスク*\n`; todayTask.forEach(t => msg += `• <@${t.User_ID}>: ${t.Task_Name}\n`); } else { msg += `※本日期限のタスクはありません。\n`; } const payload = { text: msg }; UrlFetchApp.fetch(CONFIG.SLACK_WEBHOOK_URL, { method: 'post', contentType: 'application/json', payload: JSON.stringify(payload) }); return { message: 'Slackへ通知を送信しました。' }; } // タスク追加 function addTask_(taskData) { const { sheet, headers } = getDatabase_(CONFIG.SHEET_NAME_DB); const now = new Date(); const newTaskId = 'T-' + now.getTime(); const newRow = headers.map(header => { switch(header) { case 'Task_ID': return newTaskId; case 'Tenant_ID': return taskData.tenantId || 'CLI-001'; case 'User_ID': return taskData.userId; case 'Task_Name': return taskData.taskName; case 'Deadline': return taskData.deadline; case 'Status': return taskData.status || 'today'; case 'Memo': return taskData.memo || ''; case 'Created_At':return now; default: return ''; } }); sheet.appendRow(newRow); return { message: 'タスクを追加しました', taskId: newTaskId }; } // ステータス更新 function updateTaskStatus_(taskId, newStatus) { const { tasks, sheet, headers } = getDatabase_(CONFIG.SHEET_NAME_DB); const targetIndex = tasks.findIndex(t => t.Task_ID === taskId); if (targetIndex === -1) throw new Error('タスクが見つかりません'); const statusColIdx = headers.indexOf('Status') + 1; sheet.getRange(targetIndex + 2, statusColIdx).setValue(newStatus); return { message: '更新しました' }; } // タスク更新 function updateTaskDetails_(taskData) { const { tasks, sheet, headers } = getDatabase_(CONFIG.SHEET_NAME_DB); const targetIndex = tasks.findIndex(t => t.Task_ID === taskData.taskId); if (targetIndex === -1) throw new Error('タスクが見つかりません'); const targetRowIdx = targetIndex + 2; sheet.getRange(targetRowIdx, headers.indexOf('Task_Name') + 1).setValue(taskData.taskName); sheet.getRange(targetRowIdx, headers.indexOf('User_ID') + 1).setValue(taskData.userId); sheet.getRange(targetRowIdx, headers.indexOf('Deadline') + 1).setValue(taskData.deadline); sheet.getRange(targetRowIdx, headers.indexOf('Memo') + 1).setValue(taskData.memo); return { message: '更新しました' }; }
